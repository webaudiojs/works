                    function switch1(e) {
                        if (e.target.value == "1") {
                            var osc = new DSX.Osc("sawtooth", 0);
                            osc.start();
                            adsr.soundfrom(osc, 0.1, 0.2);
                            adsr.connect(gain);
                            gain.connect(saxDelay);
                            saxDelay.connect(DAC);
                            var oscfrequency = new DSX.valueChange("knob1", osc.frequency);
                        }
